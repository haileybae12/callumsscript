        end
        return table.concat(array)

        for i,v in pairs(data) do insts[v[1]] =
Instance.new(v[2]) end
        for _,v in pairs(data) do
                for prop,val in pairs(v[3]) do

insts[val[1]]
                        else
                                insts[v[1]][prop] =
val
                        end
                end
        end

local api = {
        commands = {},
        aliases = {}
}
--[[
        @function addCommand
        @param name string
        @param cmdData table { Description, Aliases,
func }
        @desc Registers a new command and its aliases
to the system.
--]]
function api.addCommand(name, cmdData)
        local lowerName = name:lower()
        cmdData.Name = name
        api.commands[lowerName] = cmdData
        if cmdData.Aliases then
                for _, alias in
ipairs(cmdData.Aliases) do
                        api.aliases[alias:lower()] =
lowerName
                end
        end
end
--[[
        @function findCommand
        @param name string
        @desc Finds a command by its name or alias.
--]]
function api.findCommand(name)
        local lowerName = name:lower()
        local cmd = api.commands[lowerName]
        if cmd then
                return cmd
        end
        local alias = api.aliases[lowerName]
        if alias then
                return api.commands[alias]
        end
        return nil
end
--[[
return api.commands[alias]
        end
        return nil
end
--[[
        @function executeCommand
        @param commandString string
        @desc Parses and executes a command string.
--]]
function api.executeCommand(commandString)
        if not commandString or
commandString:gsub("%s", "") == "" then
                return
        end
        local args = {}
        for word in commandString:gmatch("([^%s]+)")
do
                table.insert(args, word)
        end
        local commandName = table.remove(args, 1)
        if not commandName then return end
        local command = api.findCommand(commandName)
        if command and command.func then
                -- Use pcall for robust error
handling. A failing command should not crash the
script.
                local success, err = pcall(function()
                        command.func(args)
                end)
                if not success then
                        warn("[Command Engine Error]
Failed to execute '"..commandName.."':
"..tostring(err))
                        notify("Command Error",
"Failed to execute: " .. commandName)
                end
        else
                notify("Command Error", "'" ..
commandName .. "' is not a valid command.")
        end
end

function getHierarchy(obj)
        local fullname
        local period
        if string.find(obj.Name,' ') then
                fullname = '["'..obj.Name..'"]'
                period = false
        else
                fullname = obj.Name
                period = true
        end
        local getS = obj
        local parent = obj
        local service = ''
        if getS.Parent ~= game then
                repeat
                        getS = getS.Parent
                        service = getS.ClassName
                until getS.Parent == game
        end
        if parent.Parent ~= getS then
                repeat
                        parent = parent.Parent
                        if
string.find(tostring(parent),' ') then
                                if period then
                                        fullname =
'["'..parent.Name..'"].'..fullname
                                else
                                        fullname =
'["'..parent.Name..'"]'..fullname
                                end
'["'..parent.Name..'"]'..fullname
                                end
                                period = false
                        else
                                if period then
                                        fullname =
parent.Name..'.'..fullname
                                else
                                        fullname =
parent.Name..''..fullname
                                end
                                period = true
                        end
                until parent.Parent == getS
        elseif string.find(tostring(parent),' ') then
                fullname = '["'..parent.Name..'"]'
                period = false
        end
        if period then
                return
'game:GetService("'..service..'").'..fullname
        else
                return
'game:GetService("'..service..'")'..fullname
        end
end
